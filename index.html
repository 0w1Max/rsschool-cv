const query: string = 'category=cars&color=black&year=2022';

function getParams (query: string) {
  const splitQuery = query.split('&');
  const init: any = {};

  const result = splitQuery.reduce((acc, item) => {
    const [key, value] = item.split('=');
    acc[key] = value;

    return acc;
  }, init)

  return result;
}

getParams(query);
